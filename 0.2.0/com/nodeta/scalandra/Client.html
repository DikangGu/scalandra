class Client[A, B, C](val connection : Connection, val keyspace : java.lang.String, val serializer : Serialization[A, B, C], val consistency : ConsistencyLevels)
extends Base[A, B, C] with Read[A, B, C] with Write[A, B, C] with Keyspace[A, B, C]
